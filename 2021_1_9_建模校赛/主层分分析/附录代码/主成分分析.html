# 1、记载数据，得到有用的数据
wb = open_workbook("./第三题数据.xlsx")
# sheet_names = wb.sheet_names()    读取excel文件只有一张表['Sheet1']
# 选择表
sheet = wb.sheet_by_name('Sheet1')
# xlrd直接按行读取，因此读取行数
nrows = sheet.nrows
# 创建新列表rows用来存储将写入csv的数据
rows = []
# 创建csv文件
f = open("第四题数据.csv", mode="w", newline="", encoding="utf-8-sig")
csv_write = csv.writer(f)
# 对xlrd加载的数据第一列处理
headlines = [eval(str(i).split(":")[-1]) for i in sheet.row(0)]
# 但是根据论文资料，有些列的数据也是不需要的，我们不需要的数据是：
"""
    索引列表：[0,3,5,6,8,9,11,14,18,20,25]
    分别对应食物名、硫胺素、蛋白质、核黄素、脂肪、烟酸、碳水化合物、膳食纤维、胆固醇、胡萝卜素、人民币/500g
"""
# 开始处理
needs = []
for i in range(len(headlines)):
    if i in [0,3,5,6,8,9,11,14,18,20,25]:
        pass
    else:
        needs.append(headlines[i])
csv_write.writerow(needs)

# 开始遍历读取,nrows是int类型
for i in range(1, nrows):
    # xlrd读取时，第一列和第一行前面是text:，后面是number:
    my_list = sheet.row(i)
    # 创建列表row用来对数据进行样式处理
    row = []
    # 如果第二列数据是number:0.0，则删除改行，表示该食物不吃
    if eval(str(my_list[1]).split(":")[-1]) == 0.0:
        pass
    else:
        for j in range(len(my_list)):
            if j in [0,3,5,6,8,9,11,14,18,20,25]:
                pass
            else:
                now = eval(str(my_list[j]).split(":")[-1])
                # 将处理后的数据放入行列表row中
                row.append(now)
        # 可以将数据按行形式写入csv文件了
        csv_write.writerow(row)
print("第四题数据已经加载完毕！等待使用~")
